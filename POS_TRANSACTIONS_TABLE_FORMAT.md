## Helper Function

### Payment Badge Class:
```javascript
const getPaymentBadgeClass = (method: string) => {
  const methodLower = method?.toLowerCase() || 'cash';
  if (methodLower === 'cash') return 'success';
  if (methodLower === 'card') return 'info';
  if (methodLower === 'gcash') return 'warning';
  return 'secondary';
};
```
